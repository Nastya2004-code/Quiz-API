    <script>
    class QuizApp {
        constructor() {
            this.quizContent = document.getElementById('quiz-content');
            this.errorMessage = document.getElementById('error-message');
            this.currentQuestionIndex = 0;
            this.score = 0;
            this.questions = [];

            // –°–∏–º—É–ª–∏—Ä—É–µ–º API —Å–µ—Ä–≤–µ—Ä–∞ (–≤ —Ä–µ–∞–ª—å–Ω–æ–º –ø—Ä–∏–ª–æ–∂–µ–Ω–∏–∏ –±—É–¥–µ—Ç –Ω–∞—Å—Ç–æ—è—â–∏–π fetch)
            this.serverUrl = this.mockServerUrl();
            this.fetchQuestions();
        }

        // –ò–º–∏—Ç–∞—Ü–∏—è API-—ç–Ω–¥–ø–æ–∏–Ω—Ç–∞ –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –≤–æ–ø—Ä–æ—Å–æ–≤
        mockServerUrl() {
            return {
                getQuestions: () => {
                    return new Promise((resolve) => {
                        setTimeout(() => {
                            resolve([
                                {
                                    "id": 1,
                                    "question": "–°—Ç–æ–ª–∏—Ü–∞ –†–æ—Å—Å–∏–∏?",
                                    "answers": ["–ú–æ—Å–∫–≤–∞", "–°–∞–Ω–∫—Ç-–ü–µ—Ç–µ—Ä–±—É—Ä–≥", "–ö–∞–∑–∞–Ω—å", "–ù–æ–≤–æ—Å–∏–±–∏—Ä—Å–∫"],
                                    "correctIndex": 0
                                },
                                {
                                    "id": 2,
                                    "question": "–°–∫–æ–ª—å–∫–æ –æ–∫–µ–∞–Ω–æ–≤ –Ω–∞ –ó–µ–º–ª–µ?",
                                    "answers": ["4", "5", "6", "7"],
                                    "correctIndex": 1
                                },
                                {
                                    "id": 3,
                                    "question": "–ö–∞–∫–æ–π —Å–∞–º—ã–π –±–æ–ª—å—à–æ–π –ø—Ä–æ–≥—Ä–∞–º–º–Ω—ã–π —è–∑—ã–∫?",
                                    "answers": ["Python", "Java", "JavaScript", "C++"],
                                    "correctIndex": 2
                                }
                            ]);
                        }, 1500); // –ò–º–∏—Ç–∞—Ü–∏—è –∑–∞–¥–µ—Ä–∂–∫–∏ —Å–µ—Ç–∏
                    });
                }
            };
        }

        async fetchQuestions() {
            try {
                this.questions = await this.serverUrl.getQuestions();
                this.startQuiz();
            } catch (error) {
                this.showError("–ù–µ —É–¥–∞–ª–æ—Å—å –∑–∞–≥—Ä—É–∑–∏—Ç—å –≤–æ–ø—Ä–æ—Å—ã");
            }
        }

        startQuiz() {
            this.currentQuestionIndex = 0;
            this.score = 0;
            this.renderQuestion();
        }

        renderQuestion() {
            if (this.currentQuestionIndex >= this.questions.length) {
                this.showResults();
                return;
            }

            const currentQuestion = this.questions[this.currentQuestionIndex];
            
            this.quizContent.innerHTML = `
                <h2>${currentQuestion.question}</h2>
                <div id="answers">
                    ${currentQuestion.answers.map((answer, index) => `
                        <button 
                            class="answer-btn" 
                            onclick="quizApp.checkAnswer(${index})"
                        >
                            ${answer}
                        </button>
                    `).join('')}
                </div>
            `;
        }

        checkAnswer(selectedIndex) {
            const currentQuestion = this.questions[this.currentQuestionIndex];
            const buttons = document.querySelectorAll('.answer-btn');

            buttons.forEach(button => button.disabled = true);

            if (selectedIndex === currentQuestion.correctIndex) {
                buttons[selectedIndex].classList.add('correct');
                this.score++;
            } else {
                buttons[selectedIndex].classList.add('incorrect');
                buttons[currentQuestion.correctIndex].classList.add('correct');
            }

            setTimeout(() => {
                this.currentQuestionIndex++;
                this.renderQuestion();
            }, 1000);
        }

        showResults() {
            const percentScore = (this.score / this.questions.length) * 100;
            
            this.quizContent.innerHTML = `
                <h2>–ö–≤–∏–∑ –æ–∫–æ–Ω—á–µ–Ω! üéâ</h2>
                <p>–í–∞—à —Ä–µ–∑—É–ª—å—Ç–∞—Ç: ${this.score} –∏–∑ ${this.questions.length}</p>
                <p>–ü—Ä–æ—Ü–µ–Ω—Ç –ø—Ä–∞–≤–∏–ª—å–Ω—ã—Ö –æ—Ç–≤–µ—Ç–æ–≤: ${percentScore.toFixed(2)}%</p>
                <button class="answer-btn" onclick="quizApp.startQuiz()">
                    –ü—Ä–æ–π—Ç–∏ –∫–≤–∏–∑ –∑–∞–Ω–æ–≤–æ
                </button>
            `;
        }

        showError(message) {
            this.errorMessage.textContent = message;
        }
    }

    // –°–æ–∑–¥–∞–µ–º –≥–ª–æ–±–∞–ª—å–Ω—ã–π —ç–∫–∑–µ–º–ø–ª—è—Ä –∫–≤–∏–∑–∞
    const quizApp = new QuizApp();
